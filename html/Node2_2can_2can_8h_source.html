47// CAN message data type
48// Data fields have 3 access methods (via union):
49// 8 bytes
50// 2 double words (32-bit ints)
51// 1 Byte8 dummy struct
52// The dummy struct allows for convenient construction of a CAN message from another type
53//
54// Example:
55// typedef struct {
56// uint16_t a;
57// uint8_t b;
58// float c;
59// } __attribute__((packed)) YourStruct ;
60//
61// CanMsg m = (CanMsg){
62// .id = 1,
63// .length = sizeof(YourStruct),
64// .byte8 = union_cast(Byte8, ((YourStruct){
65// .a = 10,
66// .b = 20,
67// .c = -30.0,
68// })),
69// };
70// can_printmsg(m);
71// // Should print: CanMsg(id:1, length:7, data:{10, 0, 20, 0, 0, 240, 193})
72typedef struct CanMsg CanMsg;
73struct CanMsg {
74 uint8_t id;
75 uint8_t length;
76 union {
77 uint8_t byte[8];
78 uint32_t dword[2];
79 Byte8 byte8;
80 };
81};
